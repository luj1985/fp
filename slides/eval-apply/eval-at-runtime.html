Evaluate and apply cycle

function createGenerator(seed) {
  return function() {
    var val = seed + 1;
    return {
      value: val,
      next: createGenerator(val)
    }
  }
}
var gen = createGenerator(0), limit = 0;
var current = gen();
while(limit < 10) {
  console.log(current.value);
  current = current.next();
  limit++;
}
