Imperative Programming Style

tell the computer what to do.

var arr = [1, 2, 3, 4, 5];
for (var i = 0, length = arr.length; i < length; i++) {
  var v = arr[i];
  console.log(v);
}

Functional Programming Style

describe what you want to do.

var arr = [1, 2, 3, 4, 5];
arr.forEach(function(v) {
  console.log(v);
});
